├── 📁 migrations/              # Папка автогенерируемая Alembic (миграции базы данных)

│
├── 📁 tests/                   # Папка для автоматического тестирования кода
│   ├── 📄 conftest.py

│
└── 📁 bot/                     # 🔴 ГЛАВНАЯ ПАПКА С КОДОМ БОТА
    │
    ├── 📄 __main__.py          # Главный файл запуска (точка входа: python -m bot)
    ├── 📄 config.py            # Загрузка и проверка переменных из .env (через pydantic или os)
    │
    ├── 📁 core/                # Ядро бота
    │   ├── 📄 loader.py        # Инициализация объекта бота (bot = Bot(token))
    │   └── 📄 logger.py        # Настройка красивого логирования (библиотека loguru)
    │
    ├── 📁 database/            # Работа с базой данных (обычно через ORM SQLAlchemy)
    │   ├── 📄 engine.py        # Подключение к БД
    │   ├── 📄 models.py        # Описание таблиц (class User, class Item)
    │   └── 📄 requests.py      # CRUD запросы (добавить юзера, обновить баланс)
    │
    ├── 📁 handlers/            # Обработчики сообщений (мозг)
    │   ├── 📄 __init__.py      # Здесь собираются все роутеры
    │   ├── 📄 user.py          # Команды пользователей (/start, /profile)
    │   ├── 📄 admin.py         # Команды админов (/ban, /stats)
    │   └── 📄 errors.py        # Что делать, если бот словил ошибку (чтобы не упал)
    │
    ├── 📁 middlewares/         # Промежуточные слои (срабатывают ДО обработчиков)
    │   ├── 📄 register.py      # Авто-добавление каждого написавшего в Базу Данных
    │   └── 📄 throttling.py    # Анти-спам (игнор, если юзер пишет >3 сообщений в секунду)
    │
    ├── 📁 states/              # FSM (Машина состояний) - для многошаговых диалогов
    │   ├── 📄 forms.py         # Описание шагов (например: Ввод Имени -> Ввод Возраста -> Подтверждение)
    │   └── 📄 rules.py         # Кастомные фильтры для состояний
    │
    ├── 📁 keyboards/           # Клавиатуры бота
    │   ├── 📄 inline.py        # Кнопки под сообщениями (с callback-датой)
    │   ├── 📄 reply.py         # Кнопки вместо клавиатуры (с текстом)
    │   └── 📄 builder.py       # Динамическая генерация кнопок (например, из списка товаров)
    │
    ├── 📁 services/            # Бизнес-логика и сторонние API
    │   ├── 📄 payment.py       # Интеграция с ЮKassa, Qiwi и т.д.
    │   ├── 📄 ai_helper.py     # Интеграция с ChatGPT / Qwen API
    │   └── 📄 excel_parser.py  # Работа с файлами
    │
    └── 📁 utils/               # Вспомогательные скрипты
        ├── 📄 text_format.py   # Склонение слов, форматирование дат
        └── 📄 constants.py     # Тексты сообщений (чтобы не хардкодить их в хендлерах)
